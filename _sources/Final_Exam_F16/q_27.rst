      def crypt(s, n=2, L='ABCDEFGHIJKLMNOPQRSTUVWXYZ'):
         rv = ''
         for letter in s:
            try:
               idx = L.index(letter)
               rotated_idx = (idx+n)%len(L)
               rv += L[rotated_idx]
            except:
               rv += '?'
         return rv
